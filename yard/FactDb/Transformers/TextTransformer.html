def transform(results)
  sections = []

  # Entities section
  sections << format_entities_section(results) unless results.entities.empty?

  # Facts section
  sections << format_facts_section(results) unless results.facts.empty?

  if sections.empty?
    "No results found for query: #{results.query}"
  else
    sections.join("\n\n")
  end
end
